```js
let weather = "rainy";

switch (weather) {
  case "sunny":
    console.log("It's a beautiful day!");
    break;
  case "rainy":
    console.log("Grab your umbrella");
    break;
  case "snow":
    console.log("Get your jacket");
    break;
  case "hot":
    console.log("Drink some water");
    break;
  default:
    console.log("Let's stay inside");
}
```
